%Môn: Toán
\begin{ex}
    Tính đạo hàm của $y = \cos x$ tại $x=0$.
    \shortans{0}
    \loigiai{$y' = -\sin x$. Tại $x=0$, $y' = -\sin(0) = 0$.}
\end{ex}
